[*Returns 1 if (x+y)^z = x^z+y^z and 0 if not*]
#
    integer x, y, z, tempx, tempy, tempz, tempans, ans;
    get (x, y, z);
    tempx = x;
    tempy = y;
    tempz = z;
    while (tempz > 0) {
        tempx = tempx * x;
        tempy = tempy * y;
    }
    tempans = tempx + tempy;
    tempx = x;
    while (z > 0) {
        x = x + y;
        x = x * tempx;
    }
    ans = x;
    if (ans == tempans) {
        put(1);
    } else {
        put(0);
    } endif
#
